@override

/// Default to a `http` URL
///
/// Override to generate `https` URL
Uri url(Request request) {
  var path = request.path;
  if (!path.startsWith('/')) path = '/$path';
  var version = request.version;
  if (version != null && version > 0) path = 'v$version$path';

  if (isRequestHttps) {
    return Uri.https(
      baseUrl,
      path,
      request.query,
    );
  } else {
    return Uri.http(
      baseUrl,
      path,
      request.query,
    );
  }
}
